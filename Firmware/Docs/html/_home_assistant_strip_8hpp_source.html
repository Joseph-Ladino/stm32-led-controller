74 inline void onMessage(JMQTT::Client &client,
75 const JMQTT::Message &msg) final {
76 USB_Printf("FROM LEDSTRIP: %s\n", ((string )msg).c_str());
77
78 if (msg.topic == CommandTopic) {
79 JsonDocument doc;
80 deserializeJson(doc, msg.payload);
81
82 setPower(doc["state"] == "ON");
83
84 if (doc.containsKey("color")) {
85 auto color = doc["color"];
86 if (!(color.containsKey("r") && color.containsKey("g")
87 && color.containsKey("b")))
88 return;
89
90 Color c;
91
92 c.r = color["r"];
93 c.g = color["g"];
94 c.b = color["b"];
95
96 setAll(c);
97 }
98
99 if(doc.containsKey("brightness")) {
100 setBrightness(doc["brightness"]);
101 }
102 }
103
104 publishState(&client);
105 }

107 inline void publishState(JMQTT::Client *client) final {
108 JsonDocument doc;
109
110 doc["state"] = state.powerOn ? "ON" : "OFF";
111 doc["brightness"] = state.brightness;
112 doc["effect"] = state.effect;
113 doc["color_mode"] = state.colorMode;
114 doc["color"]["r"] = state.staticColor.r;
115 doc["color"]["g"] = state.staticColor.g;
116 doc["color"]["b"] = state.staticColor.b;
117
118 string stateJson;
119 serializeJson(doc, stateJson);
120
121 USB_Printf("\nPublishing LEDStrip state to MQTT broker:\n\t%s\n\n", stateJson.c_str());
122 client->publish(StateTopic, stateJson);
123 }
