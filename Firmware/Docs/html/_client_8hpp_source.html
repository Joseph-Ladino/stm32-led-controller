77 char *curf = (char*) filter.data();
78 char *curn = (char*) topic.data();
79 char *curn_end = curn + topic.size();
80
81 while (*curf && curn < curn_end) {
82 if (*curn == '/' && *curf != '/')
83 break;
84 if (*curf != '+' && *curf != '#' && *curf != *curn)
85 break;
86 if (*curf == '+') { // skip until we meet the next separator, or end of string
87 char *nextpos = curn + 1;
88 while (nextpos < curn_end && *nextpos != '/')
89 nextpos = ++curn + 1;
90 } else if (*curf == '#')
91 curn = curn_end - 1; // skip until end of string
92 curf++;
93 curn++;
94 }
95
96 return (curn == curn_end) && (*curf == '\0');
