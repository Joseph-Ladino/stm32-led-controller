48bool PAHOClient::connect() {
49 MQTTPacket_connectData data = MQTTPacket_connectData_initializer;
50
51 data.MQTTVersion = (uint8_t) net.conf.version;
52 data.clientID.cstring = (char*) net.conf.clientName.data();
53 data.username.cstring = (char*) net.conf.username.data();
54 data.password.cstring = (char*) net.conf.password.data();
55
56 // connect socket to broker
57 if (!net.sock->isConnected()
58 && !net.sock->connectTCP(net.conf.brokerIP, net.conf.brokerPort)) {
59 return false;
60 }
61
62 // connect client (through socket) to broker
63 if (!client.connect(data) == MQTT::SUCCESS)
64 return false;
65
66 client.setDefaultMessageHandler(
67 { this, &PAHOClient::pahoMessageReceived });
68
69 if (onConnect)
70 onConnect(*this);
71
72 return true;
73}
