94 for (uint8_t b = 0; b < 8; b++) {
95 (*outPtr)[b] = ((byte << b) & 0x80) == 0 ? DMA_BIT_ZERO : DMA_BIT_ONE;

102inline void DMAStrip<_NUM_PIXELS, _NUM_PIXELS_PER_DMA, _DMA_ARR, DMA_BUF_TYPE>::onDMAInterrupt(
103 bool halfCompleteInterrupt) {
104
105 // buffer has been completely set to reset value, dma can be stopped
106 if (dmaTransferStatus == RESET_COMPLETE) {
107 dmaTransferStatus = DONE;
108 dmaStop();
109
110 frameDelay.reset();
111
112 return;
113 }
114
115 DMA_BUF_TYPE *writeBuf =
116 halfCompleteInterrupt ?
117 (DMA_BUF_TYPE*) dmaBuffer :
118 (DMA_BUF_TYPE*) (dmaBuffer + HALF_DMA_BUF_LEN);
119
120 int curLed;
121 for (curLed = 0; curLed < NUM_PIXELS_PER_DMA && dmaLedIT != dmaLedEndIT;
122 curLed++, dmaLedIT++) {
123 Color led = *dmaLedIT;
124 byteToDMATiming(led.g, &writeBuf);
125 byteToDMATiming(led.r, &writeBuf);
126 byteToDMATiming(led.b, &writeBuf);
127 }
128
129 // all data in ledstrip has been written but the DMA buffer is not full of reset data
130 if (curLed != NUM_PIXELS_PER_DMA) {
131 auto fillLeds = NUM_PIXELS_PER_DMA - curLed;
132
133 // fill with logic low value to transition to WS2815 reset period
134 memset(writeBuf, 0,
135 fillLeds * LEDBase::NUM_BYTES_PER_PIXEL * 8
136 * sizeof(DMA_BUF_TYPE));
137
138 if(fillLeds == NUM_PIXELS_PER_DMA) {
139 dmaTransferStatus = (dmaTransferStatus == RESET_STARTED) ? RESET_IN_PROGRESS : RESET_COMPLETE;
140 } else {
141 dmaTransferStatus = RESET_STARTED;
142 }
143// if(fillLeds == NUM_PIXELS_PER_DMA && i++ >= 2) dmaTransferStatus = RESET_COMPLETE;
144// else dmaTransferStatus = RESET_STARTED;
145
146// dmaTransferStatus =
147// (fillLeds == NUM_PIXELS_PER_DMA
148// && dmaTransferStatus == PARTIAL_RESET) ?
149// FULL_RESET : PARTIAL_RESET;
150 }
151}
