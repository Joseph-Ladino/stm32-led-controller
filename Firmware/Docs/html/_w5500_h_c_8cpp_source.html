58bool W5500HC::enableDHCP(uint16_t timeoutMs) {
59 if (!(initSuccess && phyLinkStatus()))
60 return false;
61
62 NetConfig netConfig = getConfig();
63
64 netConfig.mac = { 0x00, 0x08, 0xDC, 0x11, 0x22, 0x33 };
65
66 dhcpSuccess = hw.initDHCP(getFreeSocket(), &netConfig, timeoutMs);
67 if (dhcpSuccess) {
68 dhcpEnabled = true;
69 setConfig(netConfig);
70 } else {
71 dhcpEnabled = false;
72 }
73
74 return dhcpSuccess;
