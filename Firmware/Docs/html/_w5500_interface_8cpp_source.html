89bool W5500Interface::initChip() {
90 ETH_Printf("Calling wizchip_init()...\n");
91 uint8_t rx_tx_buff_sizes[2][8] = { { 2, 2, 2, 2, 2, 2, 2, 2 }, { 2, 2, 2, 2, 2, 2, 2, 2 } };
92
93 if (ctlwizchip(CW_INIT_WIZCHIP, (void*) rx_tx_buff_sizes) == -1) {
94 ETH_Printf("W5500 initialized fail.\n");
95 return false;
96 }
97
98 intr_kind intrTmp = IK_DEST_UNREACH;
99 if (ctlwizchip(CW_SET_INTRMASK, &intrTmp) == -1) {
100 ETH_Printf("W5500 interrupt\n");

102 }
103
104 return true;

107bool W5500Interface::initDHCP(W5500Socket& sock, NetConfig *conf, uint16_t timeoutMs) {
108 ETH_Printf("Calling DHCP_init()...\n");

111
112 ETH_Printf("Calling DHCP_run()...\n");
113
114 CountdownTimer timer(timeoutMs);
115 while(!(DHCP_run() == DHCP_IP_LEASED || timer.expired())) ;
116
117 if (!ipAssigned) {
118 ETH_Printf("\nIP was not assigned :(\n");
119 return false;
120 }
121
122 getIPfromDHCP(conf->ip.raw);
123 getGWfromDHCP(conf->gateway.raw);
124 getSNfromDHCP(conf->subnet.raw);
125 getDNSfromDHCP(conf->dns.raw);
126
127 printNetInfo(conf);
128

137bool W5500Interface::init(W5500Config _config) {
138 auto &inst = instance();
139 inst.config = _config;
140
141 inst.triggerReset();
142 HAL_Delay(10);
143
144 ETH_Printf("Registering W5500 callbacks...\n");
145 reg_wizchip_cs_cbfunc(enableChipSelect, disableChipSelect);
146 reg_wizchip_spi_cbfunc(readByte, writeByte);
147 reg_wizchip_spiburst_cbfunc(readBuffer, writeBuffer);
148
149 ETH_Printf("Registering DHCP callbacks...\n");
150 reg_dhcp_cbfunc(CB_DHCP_IPAssigned, CB_DHCP_IPAssigned, CB_DHCP_IPConflict);
151
152 return inst.initChip();
153}
